 async function downloadFile(url: string, folderName: string = "downloads"): Promise<string> {
        try {
            const destination = new Directory(Paths.cache, folderName);
            destination.create();

            const output = await File.downloadFileAsync(url, destination);

            if (output.exists) {
                console.log("Downloaded to:", output.uri);
                return output.uri;
            } else {
                throw new Error("File download failed");
            }
        } catch (error) {
            console.error("Download error:", error);
            throw error;

This is the api for asset fetching
export const GetAssetList = async function (
    filter: Filter[] = [],
    show_all_records = 0,
    page: number = 1,
    page_size: number = 10
) {
    const result = await clientFetch("/company/asset/list", {
        method: "POST",
        headers: {
            ...(await getAuthHeaders()),
            "Content-Type": "application/json",
        },
        body: JSON.stringify({
            page: page,
            page_size: page_size,
            show_all_records: show_all_records,
            filters: filter,
        }),
    });
    return result;
};
